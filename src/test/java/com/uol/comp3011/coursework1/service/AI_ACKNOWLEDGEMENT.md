package com.uol.comp3011.coursework1.service;

import com.uol.comp3011.coursework1.config.security.JWTUtils;
import com.uol.comp3011.coursework1.dal.entity.Role;
import com.uol.comp3011.coursework1.dal.entity.User;
import com.uol.comp3011.coursework1.dal.repository.RoleRepository;
import com.uol.comp3011.coursework1.dal.repository.UserRepository;
import com.uol.comp3011.coursework1.dto.request.UserRequest;

import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.mockito.*;

import org.springframework.security.authentication.AuthenticationManager;
import org.springframework.security.authentication.UsernamePasswordAuthenticationToken;
import org.springframework.security.core.Authentication;
import org.springframework.security.core.userdetails.UserDetails;
import org.springframework.security.crypto.bcrypt.BCryptPasswordEncoder;

import static org.junit.jupiter.api.Assertions.*;
import static org.mockito.Mockito.*;

class AuthServiceTest {

    @Mock
    private UserRepository userRepo;

    @Mock
    private RoleRepository roleRepo;

    @Mock
    private AuthenticationManager authenticationManager;

    @Mock
    private BCryptPasswordEncoder encoder;

    @Mock
    private JWTUtils jwtUtils;

    @InjectMocks
    private AuthService authService;

    @BeforeEach
    void setup() {
        MockitoAnnotations.openMocks(this);
    }

    @Test
    void createAccount_success() throws Exception {

        UserRequest req = new UserRequest("test@test.com", "password");

        when(userRepo.findByEmail(req.email())).thenReturn(null);
        when(encoder.encode(req.password())).thenReturn("hashed");
        when(roleRepo.findByName("ROLE_USER")).thenReturn(new Role());

        authService.createAccount(req);

        verify(userRepo, times(1)).save(any(User.class));
    }

    @Test
    void createAccount_duplicateUser_shouldThrowException() {

        UserRequest req = new UserRequest("test@test.com", "password");

        when(userRepo.findByEmail(req.email())).thenReturn(new User());

        assertThrows(Exception.class, () -> authService.createAccount(req));

        verify(userRepo, never()).save(any());
    }

    @Test
    void login_success_returnsToken() {

        UserRequest req = new UserRequest("test@test.com", "password");

        Authentication auth = mock(Authentication.class);
        UserDetails userDetails = mock(UserDetails.class);

        when(authenticationManager.authenticate(any(UsernamePasswordAuthenticationToken.class)))
                .thenReturn(auth);

        when(auth.getPrincipal()).thenReturn(userDetails);

        when(jwtUtils.generateToken(userDetails)).thenReturn("jwt-token");

        String token = authService.login(req);

        assertEquals("jwt-token", token);
    }
}

# 3. DataWriteService Tests

package com.uol.comp3011.coursework1.service;

import com.uol.comp3011.coursework1.dal.entity.PropertyTransaction;
import com.uol.comp3011.coursework1.dal.repository.PropertyTransactionRepository;

import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.mockito.*;

import java.io.ByteArrayInputStream;
import java.util.UUID;

import static org.mockito.Mockito.*;

class DataWriteServiceTest {

    @Mock
    private PropertyTransactionRepository repository;

    @InjectMocks
    private DataWriteService service;

    @BeforeEach
    void setup() {
        MockitoAnnotations.openMocks(this);
    }

    @Test
    void deletePpdRecord_shouldDeleteRecord() {

        UUID id = UUID.randomUUID();
        PropertyTransaction tx = new PropertyTransaction();
        tx.setTransactionUuid(id);

        when(repository.findByTransactionUuid(id)).thenReturn(tx);

        service.deletePpdRecord(id.toString());

        verify(repository).delete(tx);
    }

    @Test
    void updatePpdRecord_shouldSaveUpdatedRecord() {

        UUID id = UUID.randomUUID();

        PropertyTransaction existing = new PropertyTransaction();
        existing.setTransactionUuid(id);

        PropertyTransaction updated = new PropertyTransaction();
        updated.setTransactionUuid(id);
        updated.setPrice(100000);

        when(repository.findByTransactionUuid(id)).thenReturn(existing);

        service.updatePpdRecord(updated);

        verify(repository).save(existing);
    }

    @Test
    void uploadPpdYear_shouldSaveParsedRecords() throws Exception {

        String csv =
                "{11111111-1111-1111-1111-111111111111},100000,2023-01-01 00:00,BS1,A,N,F,1,,Street,Loc,Bristol,Dist,County,A";

        ByteArrayInputStream inputStream = new ByteArrayInputStream(csv.getBytes());

        service.uploadPpdYear(inputStream);

        verify(repository, atLeastOnce()).saveAll(any());
    }
}
