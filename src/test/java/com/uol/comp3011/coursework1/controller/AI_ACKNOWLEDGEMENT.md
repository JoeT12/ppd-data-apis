# 1. AuthController Tests

```java
package com.uol.comp3011.coursework1.controller;

import com.fasterxml.jackson.databind.ObjectMapper;
import com.uol.comp3011.coursework1.dto.request.UserRequest;
import com.uol.comp3011.coursework1.service.AuthService;
import org.junit.jupiter.api.Test;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.test.autoconfigure.web.servlet.WebMvcTest;
import org.springframework.boot.test.mock.mockito.MockBean;
import org.springframework.http.MediaType;
import org.springframework.test.web.servlet.MockMvc;

import static org.mockito.Mockito.doNothing;
import static org.mockito.Mockito.when;
import static org.springframework.test.web.servlet.request.MockMvcRequestBuilders.post;
import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.*;

@WebMvcTest(AuthController.class)
class AuthControllerTest {

    @Autowired
    private MockMvc mockMvc;

    @MockBean
    private AuthService authService;

    @Autowired
    private ObjectMapper objectMapper;

    @Test
    void createAccount_shouldReturn200() throws Exception {

        UserRequest req = new UserRequest("test@test.com", "password");

        doNothing().when(authService).createAccount(req);

        mockMvc.perform(post("/api/v1/auth/create-account")
                .contentType(MediaType.APPLICATION_JSON)
                .content(objectMapper.writeValueAsString(req)))
                .andExpect(status().isOk())
                .andExpect(jsonPath("$.message").value("Account Created Successfully"));
    }

    @Test
    void createAccount_shouldReturn400_whenMissingFields() throws Exception {

        UserRequest req = new UserRequest("", "");

        mockMvc.perform(post("/api/v1/auth/create-account")
                .contentType(MediaType.APPLICATION_JSON)
                .content(objectMapper.writeValueAsString(req)))
                .andExpect(status().isBadRequest());
    }

    @Test
    void login_shouldReturnToken() throws Exception {

        UserRequest req = new UserRequest("test@test.com", "password");

        when(authService.login(req)).thenReturn("jwt-token");

        mockMvc.perform(post("/api/v1/auth/login")
                .contentType(MediaType.APPLICATION_JSON)
                .content(objectMapper.writeValueAsString(req)))
                .andExpect(status().isOk())
                .andExpect(jsonPath("$.token").value("jwt-token"));
    }
}
```

```java
package com.uol.comp3011.coursework1.controller;

import com.uol.comp3011.coursework1.service.DataReadService;
import org.junit.jupiter.api.Test;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.test.autoconfigure.web.servlet.WebMvcTest;
import org.springframework.boot.test.mock.mockito.MockBean;
import org.springframework.security.test.context.support.WithMockUser;
import org.springframework.test.web.servlet.MockMvc;

import java.util.List;

import static org.mockito.Mockito.when;
import static org.springframework.test.web.servlet.request.MockMvcRequestBuilders.get;
import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.*;

@WebMvcTest(DataReadController.class)
class DataReadControllerTest {

    @Autowired
    private MockMvc mockMvc;

    @MockBean
    private DataReadService dataReadService;

    @Test
    @WithMockUser(roles = "USER")
    void getAvailableYears_shouldReturnYears() throws Exception {

        when(dataReadService.getPpdYearsAvailable()).thenReturn(List.of(2020,2021,2022));

        mockMvc.perform(get("/api/v1/read/data/available-years-ppd"))
                .andExpect(status().isOk())
                .andExpect(jsonPath("$[0]").value(2020));
    }

    @Test
    @WithMockUser(roles = "USER")
    void getAllRecords_shouldReturn200() throws Exception {

        when(dataReadService.getAllRecords(0,100)).thenReturn(List.of());

        mockMvc.perform(get("/api/v1/read/data/all-records"))
                .andExpect(status().isOk());
    }

    @Test
    void getAllRecords_shouldReturn403_withoutAuth() throws Exception {

        mockMvc.perform(get("/api/v1/read/data/all-records"))
                .andExpect(status().isForbidden());
    }
}
```

```java
package com.uol.comp3011.coursework1.controller;

import com.uol.comp3011.coursework1.service.DataWriteService;
import org.junit.jupiter.api.Test;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.test.autoconfigure.web.servlet.WebMvcTest;
import org.springframework.boot.test.mock.mockito.MockBean;
import org.springframework.mock.web.MockMultipartFile;
import org.springframework.security.test.context.support.WithMockUser;
import org.springframework.test.web.servlet.MockMvc;

import static org.mockito.Mockito.doNothing;
import static org.springframework.test.web.servlet.request.MockMvcRequestBuilders.multipart;
import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.status;

@WebMvcTest(DataWriteController.class)
class DataWriteControllerTest {

    @Autowired
    private MockMvc mockMvc;

    @MockBean
    private DataWriteService dataWriteService;

    @Test
    @WithMockUser(roles = "ADMIN")
    void uploadPpdYear_shouldReturn200() throws Exception {

        MockMultipartFile file =
                new MockMultipartFile("file","test.csv","text/csv","data".getBytes());

        doNothing().when(dataWriteService).uploadPpdYear(file.getInputStream());

        mockMvc.perform(multipart("/api/v1/write/admin/upload-ppd-year")
                .file(file))
                .andExpect(status().isOk());
    }

    @Test
    @WithMockUser(roles = "ADMIN")
    void uploadPpdYear_shouldReturn400_ifWrongFileType() throws Exception {

        MockMultipartFile file =
                new MockMultipartFile("file","test.txt","text/plain","data".getBytes());

        mockMvc.perform(multipart("/api/v1/write/admin/upload-ppd-year")
                .file(file))
                .andExpect(status().isBadRequest());
    }
}
```
